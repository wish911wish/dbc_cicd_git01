## 変換処理クラス

```

# common/bronze_to_silver.py（該当箇所）
class BronzeToSilverTransformer:

    def __init__(self, mapping: Mapping, expectations: Expectations,
                 silver_schema: StructType, trim_columns: list | None = None):
        self.mapping = mapping
        self.expectations = expectations
        self.silver_schema = silver_schema
        self.trim_columns = trim_columns if trim_columns is not None else mapping.source_columns()

    def check(self, df: DataFrame) -> DataFrame:
        self.mapping.validate_against(df.columns)   # 入力列の事前検証
        df = self._trim(df)
        df = self._cast(df)
        df = self._expect(df)
        return df.withColumn(
            self.CHECKS_PASSED,
            expr(f"`{self.CAST_PASSED}` AND `{self.EXPECT_PASSED}`"),
        )

    def _cast(self, df: DataFrame) -> DataFrame:
        reason_parts = []
        for cm in self.mapping:                      # ColumnMapping を反復
            df = df.withColumn(cm.typed_column, expr(cm.cast_expr()))
            reason_parts.append(
                when(expr(cm.failure_expr()), lit(cm.failure_label()))
            )
        df = df.withColumn(self.CAST_FAILED_COLUMNS, concat_ws("; ", *reason_parts))
        return df.withColumn(self.CAST_PASSED, expr(f"`{self.CAST_FAILED_COLUMNS}` = ''"))

    def _expect(self, df: DataFrame) -> DataFrame:
        if self.expectations.is_empty():
            df = df.withColumn(self.EXPECT_FAILED_RULES, lit(""))
            return df.withColumn(self.EXPECT_PASSED, lit(True))

        violation_parts = [
            when(expr(e.violation_expr()), lit(e.name)) for e in self.expectations
        ]
        df = df.withColumn(self.EXPECT_FAILED_RULES, concat_ws(", ", *violation_parts))
        return df.withColumn(self.EXPECT_PASSED, expr(f"`{self.EXPECT_FAILED_RULES}` = ''"))

    def _silver_columns(self) -> list[str]:
        return [f.name for f in self.silver_schema.fields]

```

## マッピングとExpectations

```

# common/definitions.py
"""
Bronze→Silver 変換の定義オブジェクト。
- Mapping / ColumnMapping     : 型変換仕様（業務非依存）
- Expectations / Expectation  : 業務ルール
定義の妥当性は生成時に検証し、SQL式の組み立てを各クラスに閉じ込める。
"""
from __future__ import annotations
from dataclasses import dataclass
from typing import Iterator


ALLOWED_TARGET_TYPES = {
    "INT", "BIGINT", "SMALLINT", "TINYINT",
    "DOUBLE", "FLOAT", "BOOLEAN", "DATE", "TIMESTAMP",
}
ALLOWED_FORMATS = {
    "yyyy-MM-dd", "yyyy/MM/dd", "yyyy/M/d", "yyyyMMdd",
    "yyyy-MM-dd HH:mm:ss", "yyyy/MM/dd HH:mm:ss", "yyyyMMddHHmmss",
}
CAST_PREFIX = "_cast_"


@dataclass(frozen=True)
class ColumnMapping:
    """1カラム分の型変換仕様。生成時に妥当性を検証する。"""
    source: str                 # 元カラム名
    target_type: str            # "INT" / "DECIMAL(12,2)" / "DATE" など
    format: str | None = None   # DATE/TIMESTAMP のフォーマット指定時のみ

    def __post_init__(self):
        if not self.source or not self.source.strip():
            raise ValueError("source カラム名が空です")

        t = self.target_type.upper()
        if self.format is None:
            if t not in ALLOWED_TARGET_TYPES and not t.startswith("DECIMAL("):
                raise ValueError(f"未対応の目標型です: {self.target_type}")
        else:
            if t not in ("DATE", "TIMESTAMP"):
                raise ValueError(f"format指定は DATE / TIMESTAMP のみ対応です: {self.target_type}")
            if self.format not in ALLOWED_FORMATS:
                raise ValueError(f"未許可のフォーマットです: {self.format}")

    @property
    def typed_column(self) -> str:
        """型変換結果のカラム名（例: _cast_qty）"""
        return f"{CAST_PREFIX}{self.source}"

    def cast_expr(self) -> str:
        """このカラムのキャスト SQL 式"""
        ref = f"`{self.source}`"
        t = self.target_type.upper()
        if self.format is None:
            return f"try_cast({ref} AS {self.target_type})"
        if t == "DATE":
            return f"try_to_date({ref}, '{self.format}')"
        return f"try_to_timestamp({ref}, '{self.format}')"

    def failure_expr(self) -> str:
        """変換失敗の判定式（原文字列が非NULL かつ 変換結果NULL）"""
        return f"`{self.source}` IS NOT NULL AND `{self.typed_column}` IS NULL"

    def failure_label(self) -> str:
        return f"{self.source}:cast_failed"


class Mapping:
    """ColumnMapping の集合。重複検出と反復を担う。"""

    def __init__(self, columns: list[ColumnMapping]):
        if not columns:
            raise ValueError("mapping が空です")
        sources = [c.source for c in columns]
        dupes = {s for s in sources if sources.count(s) > 1}
        if dupes:
            raise ValueError(f"source カラムが重複しています: {sorted(dupes)}")
        self._columns = list(columns)

    @classmethod
    def from_dict(cls, mapping: dict) -> "Mapping":
        """
        既存の辞書形式から生成（後方互換）。
          {"qty": "INT", "order_date": {"type": "DATE", "format": "yyyyMMdd"}}
        """
        cols = []
        for src, spec in mapping.items():
            if isinstance(spec, str):
                cols.append(ColumnMapping(source=src, target_type=spec))
            elif isinstance(spec, dict):
                cols.append(ColumnMapping(
                    source=src,
                    target_type=spec.get("type", ""),
                    format=spec.get("format"),
                ))
            else:
                raise ValueError(f"不正なマッピング指定です: {src} -> {spec}")
        return cls(cols)

    def source_columns(self) -> list[str]:
        return [c.source for c in self._columns]

    def typed_columns(self) -> list[str]:
        return [c.typed_column for c in self._columns]

    def validate_against(self, df_columns: list[str]) -> None:
        """入力 DataFrame に必要な列が揃っているかを事前検証"""
        missing = [c.source for c in self._columns if c.source not in df_columns]
        if missing:
            raise ValueError(f"入力に存在しないカラムがあります: {missing}")

    def __iter__(self) -> Iterator[ColumnMapping]:
        return iter(self._columns)

    def __len__(self) -> int:
        return len(self._columns)


@dataclass(frozen=True)
class Expectation:
    """1つの業務ルール。条件は型変換後の値（_cast_*）を参照して書く。"""
    name: str
    condition: str      # SQL ブール式

    def __post_init__(self):
        if not self.name or not self.name.strip():
            raise ValueError("ルール名が空です")
        if not isinstance(self.condition, str) or not self.condition.strip():
            raise ValueError(f"ルール条件は非空の文字列で指定してください: {self.name}")

    def violation_expr(self) -> str:
        """
        違反判定式。NULL の扱いは条件式側で担保する
        （NOT(NULL) = NULL となり違反に含まれない）。
        """
        return f"NOT ({self.condition})"


class Expectations:
    """Expectation の集合。ルール名の重複検出と反復を担う。"""

    def __init__(self, rules: list[Expectation]):
        names = [r.name for r in rules]
        dupes = {n for n in names if names.count(n) > 1}
        if dupes:
            raise ValueError(f"ルール名が重複しています: {sorted(dupes)}")
        self._rules = list(rules)

    @classmethod
    def from_dict(cls, rules: dict) -> "Expectations":
        """既存の辞書形式から生成（後方互換）。空辞書も許容（全行合格扱い）。"""
        return cls([Expectation(name=n, condition=c) for n, c in (rules or {}).items()])

    def names(self) -> list[str]:
        return [r.name for r in self._rules]

    def is_empty(self) -> bool:
        return len(self._rules) == 0

    def __iter__(self) -> Iterator[Expectation]:
        return iter(self._rules)

    def __len__(self) -> int:
        return len(self._rules)

```

## 利用側

```
mapping = Mapping([
    ColumnMapping("line_no", "INT"),
    ColumnMapping("qty", "INT"),
    ColumnMapping("unit_price", "DECIMAL(12,2)"),
    ColumnMapping("order_date", "DATE", format="yyyy/M/d"),
])

expectations = Expectations([
    Expectation("qty_positive", "_cast_qty > 0"),
    Expectation("unit_price_nonneg", "_cast_unit_price >= 0"),
    Expectation("order_date_present", "_cast_order_date IS NOT NULL"),
])

transformer = BronzeToSilverTransformer(mapping, expectations, SILVER_SCHEMA)
```
